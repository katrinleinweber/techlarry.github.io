author: larry

DFT原理

设$x(n)$为一长度为N的序列，$x(n)$的DFT变换为:

$$X(k) = sum_{0}^{N-1} x(n) W_{N}^{nk} R_{N}{(k)}$$

按时间抽选的基-2
FFT算法是DFT的一种快速算法。其难点倒位序的实现和$W_{N}{r}$的计算。

Python程序

#!/usr/bin/python
def FFT(xn):
    N = xn.__len__()
    xn = xn*(1j)
    xn = xn*(1j)
    xn = xn*(-1)
    NV2 = N/2
    J=0
    for I in xrange(N-1):
        if(I<J):
        T=xn[I]
        xn[I]=xn[J]
        xn[J]=T

        K = NV2
        while( K <= J ):
            J = J - K
            K = K/2
        J = J + K

    L = int(np.log2(N))
    for M in xrange(1,L):
        LE = np.power(2, M)
        LE1 = LE / 2

        U = 1
        W = complex(  np.cos(np.pi/LE1), np.sin(np.pi/LE1) )
        J = 0

        for J in xrange(LE1):
            I = J
            while ( I < N ):
                IP = I + LE1
                T = xn[IP]*U
                xn[IP] = xn[I] - T
                xn[I] = xn[I] + T
                I = I + LE
            U = U * W

    return xn

def FFT_show(Xk):
    """
    use matplotlib tool calculate the figure of FFT
    """

    N = Xk.__len__()
    plt.plot(np.arange(0,N)/float(N)*np.pi*2, np.abs(Xk),'o-')
    plt.xlabel(r'$\omega$')
    plt.ylabel(r'$|X(\omega)|$')
    plt.axis([0, np.pi, 0, max(np.abs(Xk))])
    plt.xticks( (0,np.pi/4,np.pi/2,np.pi*3/4, np.pi), \
        ('0',r'$\frac{\pi}{4}$', r'$\frac{\pi}{2}$',r'$\frac{3\pi}{4}$',r'$\pi$') )
    plt.show()

    if __name__ == "__main__":
        import numpy as np
        import matplotlib.pylab as plt
        n = np.linspace(-100,+100,1000)
        xn = np.sin(2*n)/np.pi/n
        xn = np.hstack((xn, np.zeros(24)))
        Xk = FFT(xn)
        FFT_show(Xk)
